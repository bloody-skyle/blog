<h2>ver.1</h2>
  
랜덤변수를 몰라서 미완성이 되고 말았다.  

```
#include<iostream>
#include<conio.h>
using namespace std;
int main(){
    char w[10][80]={};
    int key,i,j;
    int x=25;
    int y=8;
    for(i=0;i<10;i++){
        w[i][0]='*';
        w[i][79]='*';
    }
    for(i=0;i<80;i++){
        w[0][i]='*';
        w[9][i]='*';
    }
    for(i=1;i<9;i++){
        for(j=1;j<79;j++){
            w[i][j]=' ';
        }
    }
    w[8][25]='*';
    for(i=5;i<9;i++){
    	w[i][78]='@';
	}
    while(1){
        system("cls");
        cout << "미완입니다." << endl;
        for(i=0;i<10;i++){
            for(j=0;j<80;j++){
                cout << w[i][j];
            }
            cout << endl;
        }
        key=getch();
        switch(key){
            case 'w':
                if(y!=1){
                    w[y][x]=' ';
                    y=y-1;
                    w[y][x]='*';
                }
				for(i=0;i<80;i++){
            		for(j=0;j<10;j++){
            		    if(w[j][i]=='@'){
							w[j][i]=' ';
							w[j][i-1]='@';
						}
            		}
        		}
                break;
        	case 's':
        		if(y!=8){
        			w[y][x]=' ';
        			y=y+1;
        			w[y][x]='*';
				} 
				for(i=0;i<80;i++){
            		for(j=0;j<10;j++){
            		    if(w[j][i]=='@'){
							w[j][i]=' ';
							w[j][i-1]='@';
						}
            		}
        		}
		}
		if(w[y][x]=='@'){
			system("cls");
			cout << "GAME OVER";
			return 0; 
		}
    }
}
```

실행하면 턴제게임처럼 자신이 w또는 s를 눌러서 이동하면 장애물이 점점 다가온다.  

아까 말했다싶이 미완성이다.
