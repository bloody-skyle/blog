void gotoxy(int x,int y){
	COORD pos={x,y};
	SetConsoleCursorPosition(GetStdHandle(STD_OUTPUT_HANDLE), pos);
}
void exit(int status);
void check(int w[10][10]){
	int i,j,l,k;
	int c;
	for(l=0;l<10;l++){
		for(i=0;i<6;i++){
			for(j=0;j<5;j++){
				c=c+w[l][i+j];
			}
			if(c==5){
				system("cls");
				gotoxy(0,0);
				cout << "You win!";
				Sleep(2000);
				exit(0);
			}
			else if(c==-5){
				system("cls");
				gotoxy(0,0);
				cout << "You lose..";
				Sleep(2000);
				exit(0);
			}
			else{
				c=0;
			}
		}
	}
}
