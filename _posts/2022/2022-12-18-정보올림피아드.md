정보올림피아드 문제들은 다 뭣같다.

<h2>카드 바꾸기</h2>

```
//미완
#include<iostream>
using namespace std;
int n,f=1000,s=1000,l;
int card[500];
void first(){
	int i,j,plusing;
	int secard[500];
	for(i=0;i<n-1;i++){
		for(j=0;j<n;j++){
			secard[j]=card[j];
		}
		plusing=secard[i+1]-secard[i];
		for(j=i;j>0;j--){
			if(secard[j]-secard[j-1]!=plusing){
				secard[j-1]=secard[j]-plusing;
				l++;
			}
		}
		for(j=i+1;j<n-1;j++){
			if(secard[j+1]-secard[j]!=plusing){
				secard[j+1]=secard[j]+plusing;
				l++;
			}
		}
		if(l<f){
			f=l;
		}
		l=0;
	}
}
void second(){
	
}
int main(){
	int i;
	cin >> n;
	for(i=0;i<n;i++){
		cin >> card[i];
	}
	first();
	second();
	if(f<s){
		cout << f;
	}
	else{
		cout << s;
	}
}
```
